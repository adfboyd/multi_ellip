\begin{abstract}
This document records only the equations required for the current production
method used in the two-body sweep: an inviscid potential-flow boundary-element
solve coupled to a Hamiltonian midpoint step with a coupled endpoint-velocity
solve.  The method does not use the older pressure-force update, the standalone
impulse-difference update, the optional fluid-energy gradient, or the algebraic
energy projection.  Hydrodynamic energy and impulse are evaluated as state
functions of the instantaneous rigid-body state, and each time step solves for
endpoint velocities that preserve total linear impulse, total angular impulse,
and total kinetic energy.
\end{abstract}

\section{Rigid-body state}

\subsection{Continuous variables}

There are $N$ ellipsoids in an unbounded ideal fluid of density $\rho_f$.  Body
$b$ has centre $\bX_b(t)$, unit orientation quaternion $\quat{q}_b(t)$,
semi-axes $(a_b,b_b,c_b)$, solid density $\rho_{s,b}$, and volume
\begin{equation}
  V_b = \frac{4\pi}{3}a_b b_b c_b .
\end{equation}
The solid mass and body-frame inertia tensor are
\begin{equation}
  m_b = \rho_{s,b}V_b,\qquad
  \bI_b =
  \frac{m_b}{5}
  \begin{bmatrix}
    b_b^2+c_b^2 & 0 & 0\\
    0 & a_b^2+c_b^2 & 0\\
    0 & 0 & a_b^2+b_b^2
  \end{bmatrix}.
  \label{eq:mass-inertia}
\end{equation}
The lab-frame translational and angular velocities are
\begin{equation}
  \bU_b = \dot{\bX}_b,\qquad \bOmega_b \in \R^3 .
\end{equation}
The corresponding body-frame angular velocity is
\begin{equation}
  (0,\bomega_b)
  =
  \quat{q}_b^{-1}(0,\bOmega_b)\quat{q}_b .
  \label{eq:body-omega}
\end{equation}
The solid angular momentum, expressed in the lab frame, is
\begin{equation}
  \bh_{s,b}
  =
  \operatorname{vec}\!\left[
    \quat{q}_b(0,\bI_b\bomega_b)\quat{q}_b^{-1}
  \right].
  \label{eq:solid-angular-momentum}
\end{equation}
The solid kinetic energy is
\begin{equation}
  T_{s,b}
  =
  \frac{1}{2}m_b\|\bU_b\|^2
  +
  \frac{1}{2}\bomega_b^\mathsf{T}\bI_b\bomega_b .
  \label{eq:solid-ke}
\end{equation}

\subsection{Discrete state vector}

At time $t_n$ the code stores
\begin{equation}
  y_n =
  \left(
    \{\bX_b^n\}_{b=1}^N,
    \{\bU_b^n\}_{b=1}^N,
    \{\quat{q}_b^n\}_{b=1}^N,
    \{\bOmega_b^n\}_{b=1}^N
  \right).
\end{equation}
The nonlinear solve uses the endpoint generalized velocity vector
\begin{equation}
  \bz =
  \begin{bmatrix}
    \bU_1^{n+1}\\
    \vdots\\
    \bU_N^{n+1}\\
    \bOmega_1^{n+1}\\
    \vdots\\
    \bOmega_N^{n+1}
  \end{bmatrix}
  \in\R^{6N}.
  \label{eq:z-endpoint}
\end{equation}

\section{Continuous potential-flow problem}

\subsection{Governing equations}

The fluid is incompressible, inviscid, and irrotational.  Thus
\begin{equation}
  \bm{u}=\nabla\phi,\qquad
  \nabla^2\phi=0
  \quad\text{in the exterior fluid domain}.
  \label{eq:laplace}
\end{equation}
The far-field condition is
\begin{equation}
  \nabla\phi(\bx)\to\bzero
  \qquad\text{as}\qquad \|\bx\|\to\infty .
  \label{eq:far-field}
\end{equation}
On the surface $S_b$ of body $b$, the no-through-flow condition gives
\begin{equation}
  \frac{\partial\phi}{\partial n}(\bx)
  =
  \bn(\bx)\cdot
  \left[
    \bU_b+\bOmega_b\times(\bx-\bX_b)
  \right],
  \qquad \bx\in S_b .
  \label{eq:continuous-neumann}
\end{equation}

\subsection{Boundary-integral equation}

The free-space Green function is
\begin{equation}
  G(\bx,\by)=\frac{1}{4\pi\|\bx-\by\|}.
  \label{eq:green}
\end{equation}
For $\bx$ on the combined body surface $S=\cup_b S_b$, the direct boundary
integral equation is
\begin{equation}
  c(\bx)\phi(\bx)
  +
  \int_S
  \phi(\by)\frac{\partial G(\bx,\by)}{\partial n_{\by}}\dd S_{\by}
  =
  \int_S
  G(\bx,\by)\frac{\partial\phi}{\partial n}(\by)\dd S_{\by}.
  \label{eq:continuous-bie}
\end{equation}
The right-hand side is linear in the rigid-body generalized velocities, but the
operator depends on the current body configuration.

\section{Continuous energy and impulse state functions}

Given the instantaneous solution $\phi$, the fluid kinetic energy is computed
from Green's identity using the exterior-normal convention in the code:
\begin{equation}
  T_f
  =
  -\frac{\rho_f}{2}
  \int_S
  \phi\frac{\partial\phi}{\partial n}\dd S .
  \label{eq:continuous-fluid-ke}
\end{equation}
The linear Lamb impulse and angular Lamb impulse of body $b$ are
\begin{align}
  \bL_b
  &=
  \rho_f\int_{S_b}\phi\,\bn\,\dd S,
  \label{eq:continuous-lin-lamb}\\
  \bLam_b
  &=
  \rho_f\int_{S_b}
  \phi\,(\bx-\bX_b)\times\bn\,\dd S .
  \label{eq:continuous-ang-lamb}
\end{align}
These quantities are treated as state functions: no time history of $\phi$ or
finite-difference approximation to $\partial\phi/\partial t$ is used in the
current method.

The continuous conserved linear impulse, angular impulse, and Hamiltonian are
\begin{align}
  \bP
  &=
  \sum_{b=1}^N
  \left(m_b\bU_b-\bL_b\right),
  \label{eq:continuous-p}\\
  \bH
  &=
  \sum_{b=1}^N
  \left[
    \bX_b\times\left(m_b\bU_b-\bL_b\right)
    +\bh_{s,b}
    -\bLam_b
  \right],
  \label{eq:continuous-h}\\
  E
  &=
  \sum_{b=1}^N T_{s,b}+T_f .
  \label{eq:continuous-energy}
\end{align}

\section{Discrete boundary-element equations}

\subsection{Surface discretisation}

The combined surface is represented by six-node quadratic triangular elements.
Let $\phi_i$ be the unknown nodal potential and let $N_i(\xi,\eta)$ be the
quadratic element basis functions.  On element $e$,
\begin{equation}
  \bx_e(\xi,\eta)
  =
  \sum_{i\in e}N_i(\xi,\eta)\bx_i,\qquad
  \phi_e(\xi,\eta)
  =
  \sum_{i\in e}N_i(\xi,\eta)\phi_i .
\end{equation}
The oriented area vector at a quadrature point is
\begin{equation}
  \bn\,\dd S
  =
  \left(
    \frac{\partial\bx_e}{\partial\xi}
    \times
    \frac{\partial\bx_e}{\partial\eta}
  \right)
  \frac{w_q}{2},
  \label{eq:discrete-normal-area}
\end{equation}
where $w_q$ is the triangle quadrature weight.

\subsection{Discrete potential solve}

The nodal Neumann data are
\begin{equation}
  g_i
  =
  \bn_i\cdot
  \left[
    \bU_{b(i)}
    +
    \bOmega_{b(i)}\times(\bx_i-\bX_{b(i)})
  \right],
  \label{eq:discrete-neumann}
\end{equation}
where $b(i)$ is the body owning node $i$.  The BEM discretisation of
\eqref{eq:continuous-bie} gives
\begin{equation}
  \bA(\{\bX_b,\quat{q}_b\})\,\phiv
  =
  \bm{r}(\{\bX_b,\quat{q}_b\},\{\bU_b,\bOmega_b\}),
  \label{eq:discrete-bem}
\end{equation}
where $\phiv=(\phi_1,\ldots,\phi_M)^\mathsf{T}$.  Because the right-hand side
is assembled from \eqref{eq:discrete-neumann}, it is linear in the generalized
velocity at fixed configuration.

\subsection{Discrete fluid energy and impulses}

Let $Q_b$ be the set of element quadrature points owned by body $b$.  At
quadrature point $q$, let $\phi_q$ be the interpolated potential, $g_q$ the
interpolated normal derivative, $\bx_q$ the interpolated position, and
\begin{equation}
  \bm{a}_q
  =
  \left(
    \frac{\partial\bx}{\partial\xi}
    \times
    \frac{\partial\bx}{\partial\eta}
  \right)_q
  \frac{w_q}{2}
\end{equation}
the oriented area vector.  The discrete counterparts of
\eqref{eq:continuous-fluid-ke}--\eqref{eq:continuous-ang-lamb} are
\begin{align}
  T_{f,h}
  &=
  -\frac{\rho_f}{2}
  \sum_{q\in Q}
  \phi_q g_q \|\bm{a}_q\|,
  \label{eq:discrete-fluid-ke}\\
  \bL_{b,h}
  &=
  \rho_f\sum_{q\in Q_b}\phi_q\bm{a}_q,
  \label{eq:discrete-lin-lamb}\\
  \bLam_{b,h}
  &=
  \rho_f\sum_{q\in Q_b}
  \phi_q(\bx_q-\bX_b)\times\bm{a}_q .
  \label{eq:discrete-ang-lamb}
\end{align}
These are the quantities returned by the BEM impulse solve.

\section{Discrete conserved quantities}

After a BEM solve at a discrete state $y$, the discrete invariants are evaluated
as
\begin{align}
  \bP_h(y)
  &=
  \sum_{b=1}^N
  \left(m_b\bU_b-\bL_{b,h}\right),
  \label{eq:discrete-p}\\
  \bH_h(y)
  &=
  \sum_{b=1}^N
  \left[
    \bX_b\times\left(m_b\bU_b-\bL_{b,h}\right)
    +\bh_{s,b}
    -\bLam_{b,h}
  \right],
  \label{eq:discrete-h}\\
  E_h(y)
  &=
  \sum_{b=1}^N T_{s,b}+T_{f,h}.
  \label{eq:discrete-energy}
\end{align}
The output columns `pcon' and `hcon' are the per-body summands in
\eqref{eq:discrete-p} and \eqref{eq:discrete-h}.

\section{Current time step: coupled endpoint-velocity Hamiltonian solve}

\subsection{Start-of-step targets}

At the beginning of substep $n$, the solver evaluates the BEM state functions
at $y_n$ and defines the target invariant vector
\begin{equation}
  \bC^\star
  =
  \begin{bmatrix}
    \bP_h(y_n)\\
    \bH_h(y_n)
  \end{bmatrix}
  \in\R^6 .
  \label{eq:target-c}
\end{equation}
The energy target is the initial total kinetic energy of the run,
\begin{equation}
  E^\star = E_h(y_0).
  \label{eq:target-energy}
\end{equation}
Using the run initial energy rather than the previous-step energy suppresses
long-time accumulated energy drift.

\subsection{Endpoint state generated by endpoint velocities}

For a candidate endpoint velocity vector $\bz$ in \eqref{eq:z-endpoint}, the
current method constructs the endpoint state by trapezoidal kinematics:
\begin{align}
  \bX_b^{n+1}(\bz)
  &=
  \bX_b^n
  +
  \frac{h}{2}
  \left(\bU_b^n+\bU_b^{n+1}\right),
  \label{eq:endpoint-position}\\
  \bOmega_{b,m}(\bz)
  &=
  \frac{1}{2}
  \left(\bOmega_b^n+\bOmega_b^{n+1}\right),
  \label{eq:endpoint-omega-mid}\\
  \quat{q}_b^{n+1}(\bz)
  &=
  \exp\!\left[
    \frac{h}{2}(0,\bOmega_{b,m})
  \right]\quat{q}_b^n .
  \label{eq:endpoint-quat}
\end{align}
Equation \eqref{eq:endpoint-quat} is the quaternion exponential map implemented
as
\begin{equation}
  \exp\!\left[\frac{h}{2}(0,\bOmega)\right]
  =
  \left(
    \cos\frac{\|\bOmega\|h}{2},
    \frac{\sin(\|\bOmega\|h/2)}{\|\bOmega\|}\bOmega
  \right).
\end{equation}

\subsection{Endpoint residual}

The candidate endpoint state $y_{n+1}(\bz)$ is pushed into the BEM solver,
\eqref{eq:discrete-bem} is solved, and the endpoint invariants
\eqref{eq:discrete-p}--\eqref{eq:discrete-energy} are evaluated.  Define
\begin{equation}
  \bC_h(\bz)
  =
  \begin{bmatrix}
    \bP_h(y_{n+1}(\bz))\\
    \bH_h(y_{n+1}(\bz))
  \end{bmatrix}.
  \label{eq:c-of-z}
\end{equation}
The seven-equation nonlinear residual is
\begin{equation}
  \bR(\bz)
  =
  \begin{bmatrix}
    \left(\bP_h(y_{n+1}(\bz))-\bP^\star\right)/s_P\\
    \left(\bH_h(y_{n+1}(\bz))-\bH^\star\right)/s_H\\
    \left(E_h(y_{n+1}(\bz))-E^\star\right)/s_E
  \end{bmatrix}
  \in\R^7,
  \label{eq:residual}
\end{equation}
where
\begin{equation}
  s_P=\max(\|\bP^\star\|,1),\qquad
  s_H=\max(\|\bH^\star\|,1),\qquad
  s_E=\max(|E^\star|,1).
  \label{eq:residual-scales}
\end{equation}
The accepted endpoint velocity is the candidate returned after iterating toward
\begin{equation}
  \bR(\bz)=\bzero.
  \label{eq:nonlinear-system}
\end{equation}
For two bodies $\bz\in\R^{12}$ and the residual has seven components, so the
Newton correction is a minimum-norm correction of an underdetermined system.

\section{Nonlinear algebraic solve}

\subsection{Finite-difference Jacobian}

At iteration $k$, the residual Jacobian is approximated by forward differences:
\begin{equation}
  \bJ_{ij}^{(k)}
  =
  \frac{
    R_i(\bz_k+\epsilon_j\be_j)-R_i(\bz_k)
  }{\epsilon_j},
  \qquad
  \epsilon_j
  =
  \epsilon_{\mathrm{fd}}\left(|z_{k,j}|+1\right).
  \label{eq:fd-jacobian}
\end{equation}
The current sweep uses $\epsilon_{\mathrm{fd}}=10^{-3}$ and rebuilds this
Jacobian every six nonlinear iterations, with Broyden updates between rebuilds.

\subsection{Minimum-norm Newton correction}

The code solves the regularised normal equation in residual space:
\begin{equation}
  \Delta\bz_k
  =
  -\bJ_k^\mathsf{T}
  \left(
    \bJ_k\bJ_k^\mathsf{T}
    +\lambda I
  \right)^{-1}
  \bR(\bz_k),
  \qquad \lambda=10^{-10}.
  \label{eq:newton-correction}
\end{equation}
The correction is capped as
\begin{equation}
  \Delta\bz_k
  \leftarrow
  \Delta\bz_k
  \min\left(
    1,\,
    \frac{\Delta z_{\max}}{\|\Delta\bz_k\|}
  \right),
  \qquad
  \Delta z_{\max}=\frac{\delta_{\max}}{h}.
  \label{eq:step-cap}
\end{equation}
The current sweep uses $\delta_{\max}=0.2$.  Backtracking accepts the first
$\alpha\in\{1,1/2,\ldots,1/2^7\}$ satisfying
\begin{equation}
  \|\bR(\bz_k+\alpha\Delta\bz_k)\| < \|\bR(\bz_k)\|.
  \label{eq:line-search}
\end{equation}

\subsection{Broyden update}

When enabled, the finite-difference Jacobian is updated after an accepted step
by
\begin{equation}
  \bJ_{k+1}
  =
  \bJ_k
  +
  \frac{
    \left[
      \bR(\bz_{k+1})-\bR(\bz_k)-\bJ_k\Delta\bz_k
    \right]
    \Delta\bz_k^\mathsf{T}
  }{
    \Delta\bz_k^\mathsf{T}\Delta\bz_k
  }.
  \label{eq:broyden}
\end{equation}
This is the rank-one secant update used in the current sweep.

\section{Adaptive substepping}

The nominal step is $h=\Delta t$.  If the final coupled residual norm exceeds
the tolerance
\begin{equation}
  \|\bR\| > \tau,
  \qquad \tau=\texttt{hamiltonian\_floor\_tol},
  \label{eq:adaptive-test}
\end{equation}
the whole step is retried with twice as many equal substeps, up to
\begin{equation}
  n_{\mathrm{sub}}\le n_{\mathrm{sub,max}}.
  \label{eq:max-substeps}
\end{equation}
In the current sweep, $\tau=10^{-4}$ and $n_{\mathrm{sub,max}}=8$.  Each
substep applies the same endpoint-velocity solve above with $h=\Delta
t/n_{\mathrm{sub}}$.

\section{Diagnostics}

The code reports the physical conservation errors using the unscaled quantities
\begin{align}
  e_P &= \|\bP_h(y_{n+1})-\bP^\star\|,\\
  e_H &= \|\bH_h(y_{n+1})-\bH^\star\|,\\
  e_E &= \frac{|E_h(y_{n+1})-E^\star|}{\max(|E^\star|,\epsilon_{\mathrm{mach}})} .
\end{align}
The CSV output also writes, for each body,
\begin{align}
  \bp_{\mathrm{con},b}
  &=m_b\bU_b-\bL_{b,h},\\
  \bh_{\mathrm{con},b}
  &=
  \bX_b\times\bp_{\mathrm{con},b}
  +\bh_{s,b}
  -\bLam_{b,h}.
\end{align}
The sums of these per-body columns are exactly
\eqref{eq:discrete-p} and \eqref{eq:discrete-h}.  The study plots use these
columns for linear and angular impulse drift, while total kinetic-energy drift
is computed from \eqref{eq:discrete-energy}.
